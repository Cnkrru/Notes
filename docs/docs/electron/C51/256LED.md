```asm
;version：1.0
;书上原版+注释
out1 EQU P0             ;列数据高8位
out2 EQU P2             ;列数据低8位
tt   EQU P1             ;行数控制

j    EQU 30H            ;汉字索引
n    EQU 31H            ;扫描次数
i    EQU 32H            ;扫描到第几行
temp EQU 33H            ;中间数

TBL  EQU 34H            ;指针偏移低8位
TBH  EQU 35H            ;指针偏移高8位

data1 EQU 36H           ;第一个字节
data2 EQU 37H           ;第二个字节

ORG 0000H
AJMP START
ORG 0030H

;重置参数函数
START:
    MOV j,#00H      ;汉字索引初始化/重置为0
    MOV n,#00H      ;扫描次数初始化/重置为0
    mov i,#00H      ;行数初始化/重置为0

;最外层参数检测函数，j，字符数量检测
LOOP:
    CLR C           ;清空进位标志位

    MOV A,#4        ;给A字符数量值，准备比较
    
    CJNE A,j,NEXT1  ;比较A,j，不相等跳转NEXT1(嵌套1)
    AJMP START      ;A=j，说明字符展示完了，重置参数，准备下一轮循环

;第二层参数检测函数，n，扫描次数检测
NEXT1:
    CLR C           
    
    MOV A,#40       ;给A扫描次数，准备比较
    
    CJNE A,n,NEXT2  ;比较A，n，不相等跳转NEXT2（嵌套2）
    AJMP END_NEXT1  ;A=n，说明这个字扫完，跳转到下一个字（NEXT1的回调函数）

;第三层参数检测函数，i，行数检测
NEXT2:
    CLR C
    
    MOV A,#15       ;从0行开始扫描，共有16行
    
    SUBB A,i        ;A-i，计算扫到哪一行
    JC END_NEXT2    ;C=1，则跳转END_NEXT2（NEXT2的回调函数）

;工作函数，内调查表函数和延时函数
ACTION:
    MOV DPTR,#TABLE ;将指针地址指向表
    LCALL CHECK     ;调用查表函数，拿出数据到data1，data2
    
    MOV A,data1     ;将data中数据传给P0
    MOV out1,A      ;
    MOV A,data2     ;将data中数据传给P2
    MOV out2,A      ;

    ;行控制
    MOV A,i         ;行号给A
    MOV tt,i        ;行号给P1

    ACALL DELAY     ;调用延时程序，增强视觉效果

    ;防止重影
    MOV out1,#0FFH  ;将1111 1111置P0，全灭
    MOV out2,#0FFH  ;将1111 1111置P2

    INC i           ;行号+1，准备扫描下一行
    AJMP NEXT2      ;行数检测函数

;NEXT2的回调函数
END_NEXT2:
    MOV i,#00H      ;行数置0
    INC n           ;扫描次数+1
    AJMP NEXT1      ;检测扫描次数

;NEXT1的回调函数
END_NEXT1:
    MOV n,#00H      ;扫描次数置0
    INC j           ;字数+1
    AJMP LOOP       ;字数检测函数

;实现i*2+j*32，并将取表获取的数据存入data1，data2 
CHECK:
    ;计算2i
    MOV A,i         ;将字符索引给A，准备计算i*2
    RL A            ;通过左环移，在无溢出情况下，可实现i*2,比如0000 0001 -> 0000 0010 
    MOV temp,A      ;将2i暂存temp
    
    ;计算32j
    MOV A,#32       ;A=32
    MOV B,j         ;B=j
    MUL AB          ;j*32，A存低8位，B存高8位
    
    ;地址偏移传递
    CLR C 
    ADD A,temp      ;j*32+2*i(低8位)
    MOV TBL,A       ;低八位给TB-LOW
    MOV A,#00H      ;将A清零
    ADDC A,B        ;将B中32*j的高8位给A
    MOV TBH,A       ;高8位给TB-LOW
    

    CLR C

    MOV A,DPL       ;DPTR-LOW给A
    ADD A,TBL       ;低八位加和
    MOV DPL,A       ;把低八位给DPTR-LOW
    
    MOV A,DPH       ;DPTR-HIGH给A
    ADDC A,TBH      ;高8位加和
    MOV DPH,A       ;把高8位给DPTR-HIGH
    
    MOV A,#00H      ;将A清零
    MOVC A,@A+DPTR  ;查表
    MOV data1,A     ;取出第一个字节
    
    INC DPTR        ;指针+1
    MOV A,#00H      ;清零A
    MOVC A,@A+DPTR  ;查表下字节
    MOV data2,A     ;取出第二个字节
    RET

;延时程序，1ms，DELAY外层4，D0内层125，500*2us=1ms
DELAY:
    MOV R0,#4

D0:
    MOV R1,#125     
    DJNZ R1,$
    DJNZ R0,D0 
    RET

;示例数组
TABLE:
    DB 0FFH ,0BFH ,09FH ,07FH ,0DEH ,0FFH ,0D8H ,000H ,0C3H ,06DH ,0DBH ,06DH ,0DBH ,06DH ,00BH ,06DH ;
    DB 0D0H ,001H ,0DBH ,06DH ,0DBH ,06DH ,0DBH ,06DH ,0DBH ,06DH ,09BH ,0FDH ,0FFH ,0FFH ,0FFH ,0FFH ;
```
