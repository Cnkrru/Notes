## DA
1. DAC输出方式
    - 电压输出
    - 电流输出
2. 参数
    - 分辨率
        - 分辨率=满量程电压/2^N
        - N为AD/DA的位数
        - 1LSB=该位数的分辨率
    - 转换时间/转换速率
    - 转换精度
3. DAC0832
    - 8位，电流输出
    - DI0-DI7:数字信号输入端
    - CS:片选端，低电平有效
    - ILE:数据锁存允许端:高电平有效
    - WR1:写入端，低电平有效
4. 接口电路设计
    - 单缓冲
    - 双缓冲(暂时不看)
5. 波形程序
>锯齿波的产生
```asm
ORG 0000H
START:
    MOV DPTR,#0000H
    MOV A,#00H
LOOP:
    MOV @DPTR,A             ;不断把A的值给DPTR指向的外部地址
    INC A                   ;A不断自增，直到溢出，重新开始，实现锯齿波效果
    SJMP LOOP
    END
```
>三角波的产生
```asm
ORG 0000H
START:
    MOV DPTR,#0000H
    MOV A,#00H
UP:
    ;左三角
    MOVX @DPTR,A            ;不断把A的值给DPTR指向的外部地址
    INC A                   ;A不断自增

    JNZ UP                  ;未溢出，继续爬升
DOWN:
    ;右三角
    DEC A                   ;A不断自减
    MOVX @DPTR,A            ;不断把A的值给DPTR指向的外部地址

    JNZ DOWN                ;未减到0，继续减
    SJMP UP                 ;减到0，爬升
    END
```
>方波的产生
```asm
ORG 0000H
START:
    MOV DPTR,#0000H
    MOV A,#00H
LOOP:
    MOV A,#0FFH
    MOVX @DPTR,A 
    LCALL DELAY
    MOV A,#00H
    MOVX @DPTR,A 
    LCALL DELAY_IN
    SJMP LOOP
    END
DELAY:
    MOV R7,#100
DELAY_IN:
    MOV R6,#4
    DJNZ R6,$
    DJNZ R7,DELAY_IN
    RET
```
## AD
>ADC0809采样
- 把外部数据通过DPTR给A,再给30H
```asm
ORG 0000H
SJMP MAIN
ORG 0013H
SJMP INT
ORG 0030H

MAIN:
    SETB EA
    SETB EX1
    SETB IT1

    MOV DPTR,#0000H
    MOV A,#00H
    MOVX @DPTR,A
    SJMP $
INT:
    MOVX A,@DPTR
    MOV 30H,A
    RETI
```    
>采样16次，存入30H开始的16个内存地址中
```asm
ORG 0000H
SJMP MAIN
ORH 0013H
SJMP INT
ORG 0030H

MAIN:
    MOV R0,#30H
    MOV R7,#16
    SETB IT1
    SETB EA
    SETB EX1
    SJMP $
    MOV DPTR,#0002H
    MOV A.#00H
    MOVX @DPTR,A
INT:
    MOVX A,@DPTR
    MOV 30H,A
    INC R0
    DJNZ R7,LOOP
    SJMP END1
LOOP:
    MOVX A,@DPTR
END1:
    RETI
```
